<!DOCTYPE html>
<html lang="ja">
<meta charset="UTF-8">
<script>
(async () => {
  const target = "https://raw.githubusercontent.com/wista-wl-project/Wista/refs/heads/main/index.html";
  try {
    const res = await fetch(target);
    const html = await res.text();
    const w = window.open("about:blank", "_blank");
    if (!w) {
      alert("popup blocked");
      return;
    }
    w.document.open();
    w.document.write(html);
    w.document.close();
    location.replace("about:blank");
    setTimeout(() => {
      window.open('', '_self');
      window.close();
    }, 0);

  } catch(e) {
    console.error(e);
  }

})();
</script>
